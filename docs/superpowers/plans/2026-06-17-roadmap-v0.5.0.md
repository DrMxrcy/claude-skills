# roadmap v0.5.0 "control & visibility" Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add six backward-compatible features to the roadmap skill — SessionStart orientation, a Stop-hook drift nudge, `/roadmap:report` with velocity, item dependencies, `/roadmap:promote`, and `/roadmap:upgrade`.

**Architecture:** All mechanical logic lives in the single stdlib-only CLI `skills/roadmap/scripts/roadmap.py`; slash commands in `commands/roadmap/*.md` orchestrate it. New structured data lives in `.roadmap/config.json` (uniform with existing `version`/`note` fields) rather than plan frontmatter — the registry is already the structured source of truth that `status`/`render` read, so dependencies, completion dates, release dates, and the last-seen SHA all live there. `status()` computes a `blocked`/`blockedBy` view that command prompts (`next`/`build`) consume; it does not change build behaviour in Python.

**Tech Stack:** Python 3 (stdlib only; run tests with `python3.11`), bash hooks, Claude Code slash commands + hooks.

**Spec:** `docs/superpowers/specs/2026-06-17-roadmap-v0.5.0-design.md`

**Note on spec deviation:** The spec described `depends_on` in plan frontmatter. This plan stores it in the config registry (`item["dependsOn"]`) instead — same behaviour (warn+skip), but consistent with how every other structured field is stored and avoids fragile frontmatter rewriting. Display in `status`/`report` is unchanged from the spec intent.

**Conventions to match:**
- Config top-level keys are camelCase (`currentVersion`, `nextId`, `gitTagOnRelease`). New keys follow suit: `lastSeenSha`, `releases`, `skillVersion`; item keys `completedAt`, `dependsOn`.
- Versions are stored canonical (no `v`); always route new version input through the existing `_norm_version()`.
- Every mutation ends by calling `sync(root)`.
- Run the suite with `GIT_CONFIG_GLOBAL=/dev/null python3.11 -m pytest -q`.

def _record_completion(root: Path, plan_id: int) -> None:
    """Stamp/clear an item's completedAt based on whether all its steps are done."""
    cfg = read_config(root)
    for item in cfg["items"]:
        if item["id"] == plan_id:
            p = roadmap_dir(root) / item["file"]
            done, total = count_progress(p) if p.exists() else (0, 0)
            if total > 0 and done == total:

                item.pop("completedAt", None)
            write_config(root, cfg)

```python
def drift_check(root: Path) -> str | None:
    """Nudge if HEAD moved since the last check-off AND the current version is incomplete.
    Detection only — never modifies anything. Silent unless drift is likely."""
    head = _git_head(root)
    if not head:
        return None
    cfg = read_config(root)
    base = cfg.get("lastSeenSha")
    if not base or base == head:
        return None
    if not _incomplete_items(root, cfg["currentVersion"]):
        return None
    out = subprocess.run(["git", "rev-list", "--count", f"{base}..{head}"],
                         cwd=str(root), capture_output=True, text=True)
    n = out.stdout.strip() if out.returncode == 0 else "?"
    return f"⚠ {n} commit(s) since last check-off — run /roadmap:catchup?"
```

def orient_text(root: Path) -> str:
    st = status(root)
    cur = st["currentVersion"]
    items = st["items"]
    done = sum(i["done"] for i in items)
    total = sum(i["total"] for i in items)
    lines = [f"Roadmap: {st['project']} — current v{cur}",
             f"Progress: {done}/{total} steps across {len(items)} item(s)"]
    nxt = next((i for i in items
                if i["version"] == cur and not (i["total"] and i["done"] == i["total"])
                and not i.get("blocked")), None)
    if nxt:
        lines.append(f"Next unchecked: #{nxt['id']} {nxt['title']} "
                     f"[{nxt['type']}] ({nxt['done']}/{nxt['total']})")
    else:
        lines.append("No unblocked unchecked items in the current version.")
    return "\n".join(lines)


def orient_hook_json(root: Path) -> str:
    return json.dumps({"hookSpecificOutput": {
        "hookEventName": "SessionStart",
        "additionalContext": orient_text(root)}})

def set_depends(root: Path, plan_id: int, depends: list[int]) -> None:
    cfg = read_config(root)
    ids = {i["id"] for i in cfg["items"]}
    for item in cfg["items"]:
        if item["id"] == plan_id:
            if plan_id in depends:
                raise ValueError(f"item #{plan_id} cannot depend on itself")
            missing = [d for d in depends if d not in ids]
            if missing:
                raise ValueError(f"unknown dependency id(s): {missing}")
            item["dependsOn"] = depends
            write_config(root, cfg)
            sync(root)
            return
    raise ValueError(f"no plan with id {plan_id}")

def status(root: Path) -> dict:
    cfg = read_config(root)
    prog = {}
    for item in cfg["items"]:
        path = roadmap_dir(root) / item["file"]
        prog[item["id"]] = count_progress(path) if path.exists() else (0, 0)

    def done_full(iid: int) -> bool:
        d, t = prog.get(iid, (0, 0))
        return t > 0 and d == t

    items = []
    for item in cfg["items"]:
        done, total = prog[item["id"]]
        pct = round(100 * done / total) if total else 0
        blocked_by = [d for d in item.get("dependsOn", []) if not done_full(d)]
        items.append({**item, "done": done, "total": total, "pct": pct,
                      "status": derive_status(done, total),
                      "blocked": bool(blocked_by), "blockedBy": blocked_by})
    return {"project": cfg["project"], "currentVersion": cfg["currentVersion"],
            "items": items}

def write_report(root: Path) -> Path:
    st = status(root)
    cfg = read_config(root)
    releases = cfg.get("releases", {})
    by_version: dict[str, list] = {}
    for i in st["items"]:
        by_version.setdefault(i["version"], []).append(i)
    today = datetime.date.today().isoformat()
    lines = [f"# {st['project']} — Roadmap Report",
             f"_Generated {today} · current v{st['currentVersion']}_", ""]
    for v in sorted(by_version, key=_version_key):
        items = by_version[v]
        d = sum(i["done"] for i in items)
        t = sum(i["total"] for i in items)
        pct = round(100 * d / t) if t else 0
        if v in releases:
            tag = f"shipped {releases[v]}"
        elif d > 0:
            tag = "in progress"
        else:
            tag = "planned"
        filled = pct // 10
        bar = "█" * filled + "░" * (10 - filled)
        lines += [f"## v{v} — {pct}% ({tag})", f"`{bar}` {d}/{t} steps", ""]
        for i in items:
            box = "✅" if i["total"] and i["done"] == i["total"] else \
                  ("🔄" if i["done"] > 0 else "⬜")
            blk = f" — ⛔ blocked by {i['blockedBy']}" if i.get("blocked") else ""
            lines.append(f"- {box} #{i['id']} {i['title']} `{i['type']}` ({i['pct']}%){blk}")
        lines.append("")
    completed = [i for i in st["items"] if i.get("completedAt")]
    cutoff = (datetime.date.today() - datetime.timedelta(days=30)).isoformat()
    recent = [i for i in completed if i.get("completedAt", "") >= cutoff]
    lines += ["## Velocity",
              f"- Versions released: {len(releases)}",
              f"- Items completed: {len(completed)}",
              f"- Completed in last 30 days: {len(recent)}"]
    path = root / "docs" / "roadmap-report.md"
    atomic_write(path, "\n".join(lines).rstrip() + "\n")
    return path
